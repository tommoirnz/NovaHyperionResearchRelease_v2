/*
 * ============================================================
 * ATmega Quadrature Rotary Encoder + PWM H-Bridge Controller
 * ============================================================
 * Hardware:
 *   - ATmega328P (Arduino Uno compatible, 16MHz)
 *   - Rotary encoder: Channel A → PD2 (INT0), Channel B → PD3 (INT1)
 *   - H-Bridge (e.g. L298N):
 *       IN1 → PB1 (OC1A) — PWM output for forward
 *       IN2 → PB2 (OC1B) — PWM output for reverse
 *       ENA → tied HIGH (or use separate enable pin)
 *
 * Theory:
 *   Quadrature encoding uses two signals (A, B) phase-shifted 90°.
 *   By reading both on each edge, we determine direction AND speed.
 *   The encoder position drives a target speed/duty cycle for the motor.
 * ============================================================
 */

#include <avr/io.h>
#include <avr/interrupt.h>
#include <util/delay.h>
#include <stdlib.h>      /* abs() */

/* ============================================================
 * CONFIGURATION CONSTANTS
 * Tune these for your hardware setup
 * ============================================================ */

#define F_CPU           16000000UL  /* 16 MHz crystal — must match fuse bits */
#define PWM_MAX         1023        /* 10-bit PWM top value (Timer1 ICR1) */
#define PWM_MIN         0           /* Minimum duty cycle */
#define ENCODER_MAX     200         /* Max encoder count (sets speed ceiling) */
#define ENCODER_MIN    -200         /* Min encoder count (reverse ceiling) */
#define DEADBAND        5           /* Counts near zero where motor stops */
                                    /* Prevents jitter at rest position */

/* ============================================================
 * PIN DEFINITIONS
 * Using direct port manipulation — faster than Arduino digitalWrite()
 * ============================================================ */

/* Encoder inputs — must be interrupt-capable pins */
#define ENC_PORT        PORTD
#define ENC_DDR         DDRD
#define ENC_PIN         PIND
#define ENC_A           PD2         /* INT0 — triggers on any edge */
#define ENC_B           PD3         /* INT1 — triggers on any edge */

/* H-Bridge PWM outputs — Timer1 OC1A and OC1B */
#define MOTOR_DDR       DDRB
#define MOTOR_IN1       PB1         /* OC1A — forward PWM */
#define MOTOR_IN2       PB2         /* OC1B — reverse PWM */

/* ============================================================
 * GLOBAL STATE VARIABLES
 * volatile = tells compiler "this can change outside normal flow"
 * (i.e. inside an ISR — interrupt service routine)
 * Without volatile, the compiler may optimise these away — BAD.
 * ============================================================ */

volatile int16_t encoder_count = 0; /* Signed: positive=forward, negative=reverse */
volatile uint8_t last_AB = 0;       /* Previous state of A+B for direction decode */

/* ============================================================
 * QUADRATURE DECODE LOOKUP TABLE
 *
 * This is the clever bit. Instead of if/else chains,
 * we use a 16-entry lookup table indexed by:
 *   [last_A, last_B, current_A, current_B] packed into 4 bits
 *
 * Values:
 *   +1 = clockwise step
 *   -1 = counter-clockwise step
 *    0 = no movement or invalid transition
 *
 * The table encodes all valid Gray code transitions.
 * Invalid transitions (e.g. both channels changed) return 0.
 * ============================================================ */

static const int8_t QEM[16] = {
/*  curr: 00  01  10  11  */
/*  prev 00 */  0, -1, +1,  0,
/*  prev 01 */ +1,  0,  0, -1,
/*  prev 10 */ -1,  0,  0, +1,
/*  prev 11 */  0, +1, -1,  0
};
/*
 * Index calculation:
 *   index = (last_AB << 2) | current_AB
 *   last_AB  = previous (A<<1 | B)
 *   current  = current  (A<<1 | B)
 * This gives a 4-bit index into QEM[].
 */

/* ============================================================
 * INTERRUPT SERVICE ROUTINE — INT0 (Encoder Channel A)
 *
 * Fires on ANY edge of channel A (rising or falling).
 * We also read channel B here to determine direction.
 *
 * Why interrupt-driven?
 *   Polling would miss pulses at high rotation speeds.
 *   ISRs guarantee every edge is caught.
 * ============================================================ */

ISR(INT0_vect)
{
    /* Read current state of both encoder channels */
    uint8_t current_A = (ENC_PIN >> ENC_A) & 0x01;  /* Extract bit for A */
    uint8_t current_B = (ENC_PIN >> ENC_B) & 0x01;  /* Extract bit for B */

    /* Pack current state into 2 bits: [A, B] */
    uint8_t current_AB = (current_A << 1) | current_B;

    /* Build 4-bit index: upper 2 = previous state, lower 2 = current state */
    uint8_t index = (last_AB << 2) | current_AB;

    /* Look up direction from table and update count */
    encoder_count += QEM[index];

    /* Clamp to defined limits — prevents runaway accumulation */
    if (encoder_count > ENCODER_MAX) encoder_count = ENCODER_MAX;
    if (encoder_count < ENCODER_MIN) encoder_count = ENCODER_MIN;

    /* Save current state for next interrupt */
    last_AB = current_AB;
}

/* ============================================================
 * INTERRUPT SERVICE ROUTINE — INT1 (Encoder Channel B)
 *
 * Identical logic to INT0 but fires on channel B edges.
 * Using BOTH interrupts gives 4x resolution (quadrature decoding).
 * A single interrupt gives only 2x resolution.
 * ============================================================ */

ISR(INT1_vect)
{
    uint8_t current_A = (ENC_PIN >> ENC_A) & 0x01;
    uint8_t current_B = (ENC_PIN >> ENC_B) & 0x01;
    uint8_t current_AB = (current_A << 1) | current_B;
    uint8_t index = (last_AB << 2) | current_AB;

    encoder_count += QEM[index];

    if (encoder_count > ENCODER_MAX) encoder_count = ENCODER_MAX;
    if (encoder_count < ENCODER_MIN) encoder_count = ENCODER_MIN;

    last_AB = current_AB;
}

/* ============================================================
 * TIMER1 SETUP — Fast PWM, 10-bit, Phase Correct
 *
 * Timer1 is a 16-bit timer — ideal for motor PWM.
 * We use Phase-Correct PWM (mode 10) with ICR1 as TOP.
 *
 * Phase-correct PWM:
 *   Counter counts UP to ICR1, then DOWN to 0.
 *   This produces symmetric pulses — better for motors
 *   (less torque ripple vs Fast PWM).
 *
 * Frequency calculation:
 *   f_PWM = F_CPU / (2 × prescaler × ICR1)
 *   f_PWM = 16MHz / (2 × 8 × 1023) ≈ 980 Hz
 *   (audible but fine for most DC motors)
 *
 *   For 20kHz (inaudible):
 *   ICR1 = 16MHz / (2 × 1 × 20000) = 400
 *   Change prescaler to CS10 (no prescale) and ICR1=400
 * ============================================================ */

void timer1_pwm_init(void)
{
    /* Set OC1A (PB1) and OC1B (PB2) as outputs */
    MOTOR_DDR |= (1 << MOTOR_IN1) | (1 << MOTOR_IN2);

    /*
     * TCCR1A — Timer/Counter1 Control Register A
     *
     * COM1A1:0 = 10 → Clear OC1A on compare match (non-inverting)
     * COM1B1:0 = 10 → Clear OC1B on compare match (non-inverting)
     * WGM11    = 1  → Part of Mode 14 (Fast PWM, ICR1 as TOP)
     *
     * Mode 14 = Fast PWM with ICR1 as TOP:
     *   WGM13:WGM12:WGM11:WGM10 = 1:1:1:0
     */
    TCCR1A = (1 << COM1A1) | (1 << COM1B1) | (1 << WGM11);

    /*
     * TCCR1B — Timer/Counter1 Control Register B
     *
     * WGM13, WGM12 = 1 → Complete Mode 14 (Fast PWM, ICR1 TOP)
     * CS11         = 1 → Prescaler = 8
     *
     * Prescaler options:
     *   CS10 only      = /1   → 15.6 kHz PWM (with ICR1=1023)
     *   CS11 only      = /8   → 1.96 kHz PWM
     *   CS11+CS10      = /64  → 245 Hz PWM
     *   CS12 only      = /256 → 61 Hz (too slow, visible flicker)
     */
    TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS11);

    /* ICR1 = TOP value = sets PWM resolution and frequency */
    ICR1 = PWM_MAX;   /* 1023 = 10-bit resolution */

    /* Initialise both channels to 0% duty cycle (motor stopped) */
    OCR1A = 0;
    OCR1B = 0;
}

/* ============================================================
 * EXTERNAL INTERRUPT SETUP
 *
 * Configure INT0 and INT1 to fire on ANY logic change.
 * "Any change" = both rising AND falling edges.
 * This is essential for quadrature decoding — we need every edge.
 * ============================================================ */

void encoder_init(void)
{
    /* Set encoder pins as inputs */
    ENC_DDR &= ~((1 << ENC_A) | (1 << ENC_B));

    /* Enable internal pull-up resistors */
    /* Encoder outputs are open-collector — need pull-ups */
    ENC_PORT |= (1 << ENC_A) | (1 << ENC_B);

    /*
     * EICRA — External Interrupt Control Register A
     *
     * ISC00, ISC01 control INT0 trigger:
     *   00 = low level trigger
     *   01 = ANY logic change  ← we want this
     *   10 = falling edge
     *   11 = rising edge
     *
     * ISC10, ISC11 control INT1 (same options)
     *
     * Setting ISCx0=1, ISCx1=0 → trigger on any change
     */
    EICRA = (1 << ISC00) | (1 << ISC10);  /* Any change on INT0 and INT1 */

    /* EIMSK — Enable INT0 and INT1 in the interrupt mask register */
    EIMSK = (1 << INT0) | (1 << INT1);

    /* Capture initial encoder state to avoid phantom step on first interrupt */
    uint8_t init_A = (ENC_PIN >> ENC_A) & 0x01;
    uint8_t init_B = (ENC_PIN >> ENC_B) & 0x01;
    last_AB = (init_A << 1) | init_B;
}

/* ============================================================
 * SET MOTOR PWM
 *
 * Takes a signed duty value (-PWM_MAX to +PWM_MAX):
 *   Positive → forward  (OC1A active, OC1B = 0)
 *   Negative → reverse  (OC1B active, OC1A = 0)
 *   Zero     → coast    (both = 0, motor freewheels)
 *
 * H-Bridge truth table (L298N example):
 *   IN1 PWM, IN2 LOW → Forward
 *   IN1 LOW, IN2 PWM → Reverse
 *   IN1 LOW, IN2 LOW → Coast (free-wheel)
 *   IN1 HIGH,IN2 HIGH → Brake (short circuit — avoid!)
 * ============================================================ */

void motor_set(int16_t duty)
{
    if (duty > 0)
    {
        /* Forward: apply PWM to IN1, hold IN2 low */
        OCR1A = (uint16_t)duty;   /* Forward channel */
        OCR1B = 0;                /* Reverse channel off */
    }
    else if (duty < 0)
    {
        /* Reverse: hold IN1 low, apply PWM to IN2 */
        OCR1A = 0;                        /* Forward channel off */
        OCR1B = (uint16_t)(-duty);        /* Reverse channel — negate to get magnitude */
    }
    else
    {
        /* Zero: coast — both channels off */
        OCR1A = 0;
        OCR1B = 0;
    }
}

/* ============================================================
 * MAP ENCODER COUNT → PWM DUTY CYCLE
 *
 * Scales encoder_count (±ENCODER_MAX) to PWM range (0 to PWM_MAX)
 * using integer arithmetic (no floating point — saves flash/RAM).
 *
 * Formula:
 *   duty = (encoder_count × PWM_MAX) / ENCODER_MAX
 *
 * Example:
 *   encoder_count = 100, ENCODER_MAX = 200, PWM_MAX = 1023
 *   duty = (100 × 1023) / 200 = 511  (≈50% duty cycle)
 * ============================================================ */

int16_t encoder_to_pwm(int16_t count)
{
    /* Apply deadband — stops motor jittering near zero */
    if (count > -DEADBAND && count < DEADBAND)
        return 0;

    /* Scale: multiply first to preserve precision, then divide */
    return (int16_t)((int32_t)count * PWM_MAX / ENCODER_MAX);
    /*
     * Why cast to int32_t first?
     * count × PWM_MAX could be up to 200 × 1023 = 204,600
     * That overflows int16_t (max 32,767) — so we widen first.
     */
}

/* ============================================================
 * MAIN — Initialise hardware, then loop forever
 * ============================================================ */

int main(void)
{
    /* Initialise subsystems */
    encoder_init();     /* Configure encoder pins and interrupts */
    timer1_pwm_init();  /* Configure Timer1 for PWM output */

    /* Enable global interrupts — ISRs won't fire without this! */
    sei();

    /* --------------------------------------------------------
     * Main control loop
     *
     * Reads encoder count (set by ISRs asynchronously),
     * maps it to a PWM duty cycle, and drives the H-bridge.
     *
     * The ISRs update encoder_count independently of this loop.
     * This is a simple open-loop speed control:
     *   "Encoder position → motor speed"
     *
     * For closed-loop (PID), you'd add:
     *   - A tachometer or second encoder for feedback
     *   - PID error calculation here
     *   - See our earlier PID discussion!
     * -------------------------------------------------------- */

    while (1)
    {
        /* Safely read volatile encoder_count */
        /* Disable interrupts briefly to get atomic 16-bit read */
        cli();                              /* Disable interrupts */
        int16_t count = encoder_count;     /* Atomic copy */
        sei();                              /* Re-enable interrupts */
        /*
         * Why atomic read?
         * encoder_count is 16-bit. On an 8-bit AVR, reading 16 bits
         * takes two instructions. An ISR could fire between them,
         * corrupting the value. cli/sei brackets prevent this.
         */

        /* Map encoder position to PWM duty */
        int16_t duty = encoder_to_pwm(count);

        /* Drive the H-bridge */
        motor_set(duty);

        /* Small delay — tune to taste */
        /* Shorter = more responsive, longer = smoother */
        _delay_ms(10);
    }

    return 0;  /* Never reached, but good practice */
}
