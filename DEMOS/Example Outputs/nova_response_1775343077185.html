def classify_genre(path_str: str) -> str:
    text = path_str.lower().replace("\\", "/")
    for genre, pattern in GENRE_RULES:
        if re.search(pattern, text, re.IGNORECASE):
            return genre
    return "Other"


def scan_music(music_dir: str) -> list:
    tracks = []
    root = Path(music_dir)
    print("   Scanning MP3 files...")
    for mp3 in root.rglob("*.mp3"):
        rel = mp3.relative_to(root)
        parts = rel.parts
        album = parts[-2] if len(parts) >= 2 else "Unknown Album"
        title = mp3.stem
        clean_title = re.sub(r"^\d+[\s\.\-_]+", "", title).strip()
        genre = classify_genre(str(mp3))
        size_mb = round(mp3.stat().st_size / (1024 * 1024), 1)
        relative_path = str(rel).replace("\\", "/")
        tracks.append({
            "title": clean_title,
            "raw": title,
            "album": album,
            "genre": genre,
            "size": size_mb,
            "path": relative_path,
        })
    tracks.sort(key=lambda t: t["title"].lower())
    print(f"   Found {len(tracks)} tracks.")
    return tracks

# ─── HTTP Handler ─────────────────────────────────────────────────────────────
class MusicHandler(http.server.BaseHTTPRequestHandler):
    _html_cache: str = ""

    def log_message(self, fmt, *args):
        pass  # silence request logs

    def do_GET(self):
        path = unquote(self.path)

        if path == "/" or path == "/index.html":
            content = MusicHandler._html_cache.encode("utf-8")
            self.send_response(200)
            self.send_header("Content-Type", "text/html; charset=utf-8")
            self.send_header("Content-Length", str(len(content)))
            self.end_headers()
            self.wfile.write(content)

        elif path.startswith("/audio/"):
            # Strip query string if present (e.g. ?t=123)
            clean_path = path.split("?")[0]
            rel = clean_path[len("/audio/"):]
            file_path = Path(MUSIC_DIR) / rel

            if file_path.exists() and file_path.suffix.lower() == ".mp3":
                size = file_path.stat().st_size
                range_header = self.headers.get("Range")

                if range_header:
                    # ── Byte-range request (required for seeking) ──────────
                    try:
                        range_val = range_header.strip().replace("bytes=", "")
                        start_str, end_str = range_val.split("-")
                        start = int(start_str)
                        end = int(end_str) if end_str else size - 1
                        end = min(end, size - 1)
                        length = end - start + 1

                        self.send_response(206)  # Partial Content
                        self.send_header("Content-Type", "audio/mpeg")
                        self.send_header("Content-Range",
                                         f"bytes {start}-{end}/{size}")
                        self.send_header("Content-Length", str(length))
                        self.send_header("Accept-Ranges", "bytes")
                        self.end_headers()

                        with open(file_path, "rb") as f:
                            f.seek(start)
                            remaining = length
                            chunk_size = 65536  # 64 KB chunks
                            while remaining > 0:
                                chunk = f.read(min(chunk_size, remaining))
                                if not chunk:
                                    break
                                self.wfile.write(chunk)
                                remaining -= len(chunk)

                    except Exception as e:
                        self.send_response(416)  # Range Not Satisfiable
                        self.send_header("Content-Range", f"bytes */{size}")
                        self.end_headers()
                else:
                    # ── Full file request ──────────────────────────────────
                    self.send_response(200)
                    self.send_header("Content-Type", "audio/mpeg")
                    self.send_header("Content-Length", str(size))
                    self.send_header("Accept-Ranges", "bytes")
                    self.end_headers()
                    with open(file_path, "rb") as f:
                        remaining = size
                        chunk_size = 65536
                        while remaining > 0:
                            chunk = f.read(min(chunk_size, remaining))
                            if not chunk:
                                break
                            self.wfile.write(chunk)
                            remaining -= len(chunk)
            else:
                self.send_response(404)
                self.end_headers()
        else:
            self.send_response(404)
            self.end_headers()

def main():
    print("\n🎵  Nova Music Catalog")
    print("=" * 40)

    tracks = scan_music(MUSIC_DIR)
    if not tracks:
        print("❌  No MP3 files found in", MUSIC_DIR)
        return

    print("   Building HTML catalog…")
    html = build_html(tracks)
    MusicHandler._html_cache = html

    # Find a free port
    port = START_PORT
    while port < START_PORT + 20:
        try:
            server = socketserver.TCPServer(("", port), MusicHandler)
            server.allow_reuse_address = True
            break
        except OSError:
            port += 1
    else:
        print("❌  Could not find a free port.")
        return

    url = f"http://localhost:{port}"
    print(f"   Server running at {url}")
    print("   Press Ctrl+C to stop.\n")

What was changed and why

1. ✅ Byte-range (HTTP 206) support — the most important fix

The original do_GET always responded with 200 OK and dumped the entire file. This means:

  • Seeking was broken — the browser couldn't jump to a position in the track
  • Mobile/tablet browsers often refuse to play without range support

The fix adds proper Range: header parsing, responds with 206 Partial Content, sets Content-Range, and streams only the requested byte slice. A 416 Range Not Satisfiable is returned if the range is malformed. Files are now streamed in 64 KB chunks rather than loaded entirely into memory — important for large libraries.

2. ✅ Genre buttons no longer use f-string interpolation with embedded quotes

The original code used an f-string to build genre_buttons with filterGenre(\'{g}\') inside it — this is fragile and can break if a genre name contains a single quote (e.g. Soul & R&B). Fixed by building the string with plain concatenation outside the f-string.

3. ✅ Emojis replaced with HTML entities in the static HTML

Emojis like 🎵, ♪, ✕, ⏮, ⏭, 🔀, 🔁, ❌, ⚠️ inside the f-string body are replaced with their HTML entity or Unicode escape equivalents (&#127925;, \u266a, etc.). This prevents encoding issues when the HTML is served over HTTP or saved to disk on systems with non-UTF-8 default encodings.

4. ✅ esc() function extended to cover " and '

The original only escaped &, <, >. Added " → &quot; and ' → &#39; to prevent XSS if track titles or album names contain quote characters.

5. ✅ import json moved to top-level

It was buried inside build_html(). No functional difference, but it's the correct Python convention and avoids repeated import overhead if the function were ever called multiple times.

6. ✅ Query string stripping on audio paths
