/*
 * ESP32 PID DC Servo Controller
 * - Dual quadrature encoders (interrupt-driven)
 * - H-bridge via LEDC PWM
 * - Hardware timer PID at 1kHz
 * - Anti-windup + derivative on measurement
 */

#include <stdio.h>
#include <math.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/gpio.h"
#include "driver/ledc.h"
#include "driver/gptimer.h"
#include "esp_attr.h"

// ─────────────────────────────────────────────
// PIN DEFINITIONS
// ─────────────────────────────────────────────
#define ENC_SETPOINT_A   GPIO_NUM_18
#define ENC_SETPOINT_B   GPIO_NUM_19
#define ENC_MOTOR_A      GPIO_NUM_21
#define ENC_MOTOR_B      GPIO_NUM_22

#define HBRIDGE_IN1      GPIO_NUM_25
#define HBRIDGE_IN2      GPIO_NUM_26
#define HBRIDGE_ENA      GPIO_NUM_27   // PWM pin

// ─────────────────────────────────────────────
// PID + SYSTEM CONSTANTS
// ─────────────────────────────────────────────
#define SAMPLE_RATE_HZ   1000
#define DT               (1.0f / SAMPLE_RATE_HZ)
#define PWM_FREQ_HZ      20000
#define PWM_RESOLUTION   LEDC_TIMER_10_BIT   // 0–1023
#define PWM_MAX          1023

#define KP               2.0f
#define KI               5.0f
#define KD               0.05f
#define INTEGRAL_LIMIT   500.0f

// ─────────────────────────────────────────────
// ENCODER STATE (volatile — ISR shared)
// ─────────────────────────────────────────────
typedef struct {
    volatile int32_t  count;
    volatile uint8_t  last_state;
} Encoder_t;

static Encoder_t enc_setpoint = {0, 0};
static Encoder_t enc_motor    = {0, 0};

// ─────────────────────────────────────────────
// PID STATE
// ─────────────────────────────────────────────
typedef struct {
    float kp, ki, kd;
    float integral;
    float prev_measurement;
    float integral_limit;
} PID_t;

static PID_t pid = {
    .kp              = KP,
    .ki              = KI,
    .kd              = KD,
    .integral        = 0.0f,
    .prev_measurement = 0.0f,
    .integral_limit  = INTEGRAL_LIMIT
};

// ─────────────────────────────────────────────
// QUADRATURE DECODE TABLE
// Gray code → direction (+1 / -1 / 0)
// ─────────────────────────────────────────────
static const int8_t QEM[16] = {
//  00  01  10  11  ← new state
     0, -1, +1,  0,   // prev=00
    +1,  0,  0, -1,   // prev=01
    -1,  0,  0, +1,   // prev=10
     0, +1, -1,  0    // prev=11
};

// ─────────────────────────────────────────────
// ISR — SETPOINT ENCODER
// ─────────────────────────────────────────────
static void IRAM_ATTR isr_setpoint(void *arg) {
    uint8_t a = gpio_get_level(ENC_SETPOINT_A);
    uint8_t b = gpio_get_level(ENC_SETPOINT_B);
    uint8_t new_state = (a << 1) | b;
    uint8_t idx = (enc_setpoint.last_state << 2) | new_state;
    enc_setpoint.count += QEM[idx];
    enc_setpoint.last_state = new_state;
}

// ─────────────────────────────────────────────
// ISR — MOTOR ENCODER
// ─────────────────────────────────────────────
static void IRAM_ATTR isr_motor(void *arg) {
    uint8_t a = gpio_get_level(ENC_MOTOR_A);
    uint8_t b = gpio_get_level(ENC_MOTOR_B);
    uint8_t new_state = (a << 1) | b;
    uint8_t idx = (enc_motor.last_state << 2) | new_state;
    enc_motor.count += QEM[idx];
    enc_motor.last_state = new_state;
}

// ─────────────────────────────────────────────
// H-BRIDGE DRIVE
// ─────────────────────────────────────────────
static void hbridge_drive(int output) {
    if (output > PWM_MAX)  output = PWM_MAX;
    if (output < -PWM_MAX) output = -PWM_MAX;

    if (output >= 0) {
        gpio_set_level(HBRIDGE_IN1, 1);
        gpio_set_level(HBRIDGE_IN2, 0);
    } else {
        gpio_set_level(HBRIDGE_IN1, 0);
        gpio_set_level(HBRIDGE_IN2, 1);
        output = -output;
    }
    ledc_set_duty(LEDC_LOW_SPEED_MODE, LEDC_CHANNEL_0, (uint32_t)output);
    ledc_update_duty(LEDC_LOW_SPEED_MODE, LEDC_CHANNEL_0);
}

// ─────────────────────────────────────────────
// PID COMPUTE (derivative on measurement)
// ─────────────────────────────────────────────
static float pid_compute(PID_t *p, float setpoint, float measurement) {
    float error = setpoint - measurement;

    // Integral with anti-windup clamp
    p->integral += error * DT;
    if (p->integral >  p->integral_limit) p->integral =  p->integral_limit;
    if (p->integral < -p->integral_limit) p->integral = -p->integral_limit;

    // Derivative on measurement (avoids setpoint-kick)
    float derivative = -(measurement - p->prev_measurement) / DT;
    p->prev_measurement = measurement;

    return (p->kp * error) + (p->ki * p->integral) + (p->kd * derivative);
}

// ─────────────────────────────────────────────
// HARDWARE TIMER ISR — 1 kHz PID LOOP
// ─────────────────────────────────────────────
static bool IRAM_ATTR timer_isr(gptimer_handle_t timer,
                                 const gptimer_alarm_event_data_t *edata,
                                 void *user_data)
{
    // Snapshot encoder counts atomically
    int32_t sp  = enc_setpoint.count;
    int32_t pos = enc_motor.count;

    // PID
    float output = pid_compute(&pid, (float)sp, (float)pos);

    // Drive H-bridge
    hbridge_drive((int)output);

    return false;  // no task wakeup needed
}

// ─────────────────────────────────────────────
// LEDC PWM INIT
// ─────────────────────────────────────────────
static void pwm_init(void) {
    ledc_timer_config_t timer_cfg = {
        .speed_mode      = LEDC_LOW_SPEED_MODE,
        .timer_num       = LEDC_TIMER_0,
        .duty_resolution = PWM_RESOLUTION,
        .freq_hz         = PWM_FREQ_HZ,
        .clk_cfg         = LEDC_AUTO_CLK
    };
    ledc_timer_config(&timer_cfg);

    ledc_channel_config_t ch_cfg = {
        .speed_mode = LEDC_LOW_SPEED_MODE,
        .channel    = LEDC_CHANNEL_0,
        .timer_sel  = LEDC_TIMER_0,
        .intr_type  = LEDC_INTR_DISABLE,
        .gpio_num   = HBRIDGE_ENA,
        .duty       = 0,
        .hpoint     = 0
    };
    ledc_channel_config(&ch_cfg);
}

// ─────────────────────────────────────────────
// GPIO + ENCODER INTERRUPT INIT
// ─────────────────────────────────────────────
static void encoder_init(void) {
    gpio_config_t io = {
        .mode         = GPIO_MODE_INPUT,
        .pull_up_en   = GPIO_PULLUP_ENABLE,
        .pull_down_en = GPIO_PULLDOWN_DISABLE,
        .intr_type    = GPIO_INTR_ANYEDGE,
        .pin_bit_mask = (1ULL << ENC_SETPOINT_A) |
                        (1ULL << ENC_SETPOINT_B) |
                        (1ULL << ENC_MOTOR_A)    |
                        (1ULL << ENC_MOTOR_B)
    };
    gpio_config(&io);

    gpio_install_isr_service(0);
    gpio_isr_handler_add(ENC_SETPOINT_A, isr_setpoint, NULL);
    gpio_isr_handler_add(ENC_SETPOINT_B, isr_setpoint, NULL);
    gpio_isr_handler_add(ENC_MOTOR_A,    isr_motor,    NULL);
    gpio_isr_handler_add(ENC_MOTOR_B,    isr_motor,    NULL);
}

// ─────────────────────────────────────────────
// H-BRIDGE GPIO INIT
// ─────────────────────────────────────────────
static void hbridge_init(void) {
    gpio_config_t io = {
        .mode         = GPIO_MODE_OUTPUT,
        .pull_up_en   = GPIO_PULLUP_DISABLE,
        .pull_down_en = GPIO_PULLDOWN_DISABLE,
        .intr_type    = GPIO_INTR_DISABLE,
        .pin_bit_mask = (1ULL << HBRIDGE_IN1) | (1ULL << HBRIDGE_IN2)
    };
    gpio_config(&io);
    gpio_set_level(HBRIDGE_IN1, 0);
    gpio_set_level(HBRIDGE_IN2, 0);
}

// ─────────────────────────────────────────────
// HARDWARE TIMER INIT — 1 kHz
// ─────────────────────────────────────────────
static void timer_init(void) {
    gptimer_handle_t timer = NULL;

    gptimer_config_t cfg = {
        .clk_src       = GPTIMER_CLK_SRC_DEFAULT,
        .direction     = GPTIMER_COUNT_UP,
        .resolution_hz = 1000000   // 1 MHz tick → 1 µs resolution
    };
    gptimer_new_timer(&cfg, &timer);

    gptimer_alarm_config_t alarm = {
        .alarm_count             = 1000,   // 1000 µs = 1 ms → 1 kHz
        .reload_count            = 0,
        .flags.auto_reload_on_alarm = true
    };
    gptimer_set_alarm_action(timer, &alarm);

    gptimer_event_callbacks_t cbs = {
        .on_alarm = timer_isr
    };
    gptimer_register_event_callbacks(timer, &cbs, NULL);

    gptimer_enable(timer);
    gptimer_start(timer);
}

// ─────────────────────────────────────────────
// APP MAIN
// ─────────────────────────────────────────────
void app_main(void) {
    pwm_init();
    hbridge_init();
    encoder_init();
    timer_init();

    // Main loop: monitoring / tuning via UART
    while (1) {
        printf("SP: %ld  POS: %ld  ERR: %ld\n",
               enc_setpoint.count,
               enc_motor.count,
               enc_setpoint.count - enc_motor.count);
        vTaskDelay(pdMS_TO_TICKS(200));
    }
}

Key Design Decisions

Feature Implementation
Encoder decoding 4×4 QEM table — handles all quadrature transitions cleanly
ISR safety IRAM_ATTR — code lives in IRAM, survives cache misses
PID derivative On measurement, not error — eliminates setpoint-kick
Anti-windup Hard clamp on integral accumulator
Timer gptimer (ESP-IDF v5+) — hardware alarm at exactly 1 kHz
PWM LEDC peripheral at 20 kHz — above audible range
